\textbf{PROBLEM}:-Sachin invests \rupee$8500$ in 10\%, \rupee100 shares at \rupee170. He sells the shares when the price of each share rises by \rupee30. He invests the proceeds in 12\%, \rupee100 shares at \rupee$125$.\\
Find:\\
(i) the sale proceeds.\\
(ii) the number of \rupee125 shares he buys.\\
(iii) the change in his annual income.\\

\bigskip

\textbf{SOLUTION}:-\\
given\\
\hspace*{1cm} investment = \rupee$8500$\\
\hspace*{1cm} face value = \rupee$100$\\
\hspace*{1cm} shareprice of $1$st company = \rupee$170$\\
\hspace*{1cm} shareprice of $2$nd company = \rupee$125$\\
\hspace*{1cm} profit = \rupee$30$\\
\hspace*{1cm} percentage of $1$st company = $10\%$\\
\hspace*{1cm} percentage of $2$nd company = $12\%$\\

\bigskip

\textbf{(i)}
sales proceeds = $\text{shares} \times \text{(share price + profit)}$\\
\begin{align*}
\text{no of shares}&= \frac{\text{investment}}{\text{share price}}\\
     &= \frac{8500}{170}\\
     &=50\\
     \text{sale proceeds}&= 50 \times(170+30)\\
     &=10,000\\
\end{align*}

\bigskip

\textbf{(ii)}
new number of shares
\begin{align*}
&=\frac{\text{sale proceeds}}{\text{share price}}\\
&=\frac{10,000}{125}\\
\text{new shares}&=80
\end{align*}

\bigskip

\textbf{(iii)} 
change in income= new dividend - old dividend\\
\begin{align*}
\text{dividend}&=\frac{\text{shares} \times \text{face value} \times \text{percentage}}{100}\\
\text{old dividend}&=\frac{50 \times 100 \times 10}{100}\\
&=500\\
\text{new dividend}&=\frac{80 \times 100 \times 12}{100}\\
&=960\\
\text{change in income}&=960-500\\
&=460.
\end{align*}
